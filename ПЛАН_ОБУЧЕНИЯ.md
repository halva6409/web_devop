    async def confirm_add(self, update: Update, context: ContextTypes.DEFAULT_TYPE):

                f'{API_BASE_URL}/news',

                    'title': context.user_data['title'],
                    'source': context.user_data['source'],
                    'content': context.user_data['content']

            if response.status_code == 201:
                await query.edit_message_text("‚úÖ –ù–æ–≤–æ—Å—Ç—å —É—Å–ø–µ—à–Ω–æ –¥–æ–±–∞–≤–ª–µ–Ω–∞ –≤ –ë–î!")

                await query.edit_message_text(f"‚ùå –û—à–∏–±–∫–∞: {response.text}")
        
        except Exception as e:

            logging.error(f"Error: {str(e)}")
        
        return ConversationHandler.END
    
    async def cancel_add(self, update: Update, context: ContextTypes.DEFAULT_TYPE):
        """–û—Ç–º–µ–Ω–∞ –¥–æ–±–∞–≤–ª–µ–Ω–∏—è"""

        if query:
            await query.answer()
            await query.edit_message_text("‚ùå –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –æ—Ç–º–µ–Ω–µ–Ω–æ")
        else:
            await update.message.reply_text("‚ùå –î–æ–±–∞–≤–ª–µ–Ω–∏–µ –æ—Ç–º–µ–Ω–µ–Ω–æ")
        return ConversationHandler.END
    
    async def list_news(self, update: Update, context: ContextTypes.DEFAULT_TYPE):
        """–ü–æ–∫–∞–∑–∞—Ç—å –≤—Å–µ –Ω–æ–≤–æ—Å—Ç–∏"""

            response = requests.get(f'{API_BASE_URL}/news', timeout=5)
            
            if response.status_code == 200:
                news_list = response.json()
                
                if not news_list:
                    await update.message.reply_text("üì≠ –ù–æ–≤–æ—Å—Ç–µ–π –Ω–µ—Ç")

                # –ü–æ–∫–∞–∑–∞—Ç—å –ø–æ 3 –Ω–æ–≤–æ—Å—Ç–∏ –∑–∞ —Ä–∞–∑

                    message += f"<b>{news['title']}</b>\n"
                    message += f"–ò—Å—Ç–æ—á–Ω–∏–∫: {news['source']}\n"

                        message += f"{news['content'][:100]}...\n\n"
                
                await update.message.reply_text(message, parse_mode='HTML')
            else:
                await update.message.reply_text("‚ùå –û—à–∏–±–∫–∞ –ø–æ–ª—É—á–µ–Ω–∏—è –Ω–æ–≤–æ—Å—Ç–µ–π")
        
        except Exception as e:
            await update.message.reply_text(f"‚ùå –û—à–∏–±–∫–∞: {str(e)}")
            logging.error(f"Error: {str(e)}")
    
    def run(self):
